POST api.nono.fi:5000/profile
{
    "firstname": "weicheng",
    "lastname": "Zou",
    "birthYear": 1995,
    "gender": 1,
    "phone": "0436333333"
}
type:
string: firstname, lastname, phone 
int: birthYear, gender 
response code: 
401 wrong password, 200 success, 400 not receive enough attributes 

{
    "firstname": "weicheng",
    "lastname": "Zou",
    "birthYear": 1995,
    "gender": 1,
    "phone": "0436333333"
    "bidderTag": 1,
    "bsb": 155325,
    "accNumber": 123456789,
    "initialBid": 11111
}
type:
string: firstname, lastname, phone
int: birthYear, gender, bidderTag, bsb, accNumber, initialBid
response code: 
401 wrong password, 200 success, 400 not receive enough attributes
note:
if bidderTag = 0, will ignore bsb, accNumber, initialBid even if they are posted
if bidderTag = 1, will need bsb, accNumber, initialBid or will throw error 400
